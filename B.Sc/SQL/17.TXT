CREATE TABLE BRANCH(BRANCH_IFSC NUMBER(2) PRIMARY KEY,
BRANCH_NAME VARCHAR2(8),
BRANCH_CITY VARCHAR2(8),
ASSETS NUMBER(6));


CREATE TABLE LOAN (LOAN_ID NUMBER(4) PRIMARY KEY,
BRANCH_ID NUMBER(2),
AMMOUNT NUMBER(8),
FOREIGN KEY(BRANCH_ID)REFERENCES BRANCH(BRANCH_IFSC));


CREATE TABLE BORROWER(CUST_NAME VARCHAR2(8),
LOAN_ID NUMBER(4),
FOREIGN KEY(LOAN_ID) REFERENCES LOAN(LOAN_ID));
